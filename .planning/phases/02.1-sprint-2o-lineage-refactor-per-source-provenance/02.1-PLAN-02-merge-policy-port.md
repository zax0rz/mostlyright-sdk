```python
class _ReverseStr:
    """Wraps a string so that comparison is reversed.

    `min([_ReverseStr("a"), _ReverseStr("b")]) == _ReverseStr("b")` because the
    underlying comparison is inverted. Used to make `min()` select the LATEST
    RFC3339 timestamp when the sort key calls for "most recent wins."
    """
    __slots__ = ("value",)
    def __init__(self, value: str) -> None: self.value = value
    def __lt__(self, other: "_ReverseStr") -> bool: return self.value > other.value
    def __eq__(self, other: object) -> bool:
        return isinstance(other, _ReverseStr) and self.value == other.value
    def __hash__(self) -> int: return hash((self.__class__, self.value))
    def __repr__(self) -> str: return f"_ReverseStr({self.value!r})"
```

```python
from collections.abc import Mapping
from dataclasses import dataclass, field
from types import MappingProxyType
from typing import Callable

Selector = Callable[..., dict | None]

@dataclass(frozen=True)
class MergePolicy:
    name: str
    source_priority: Mapping[str, int]
    parser_filter: frozenset[str]
    parser_tiebreak: Mapping[str, int]
    selector: Selector
    def __post_init__(self):
        # Reject plain dicts to prevent runtime mutation of priority maps
        for attr in ("source_priority", "parser_tiebreak"):
            v = getattr(self, attr)
            if not isinstance(v, MappingProxyType):
                raise TypeError(f"{attr} must be MappingProxyType, got {type(v).__name__}")

def first_reported_highest_priority_with_received_at_tiebreak(
    rows: list[dict],
    *,
    source_priority: Mapping[str, int],
    parser_filter: frozenset[str],
    parser_tiebreak: Mapping[str, int],
) -> dict | None:
    candidates = [
        r for r in rows
        if (r.get("source") or "") in source_priority
        and (r.get("parser_name") or "mostlyright_v1") in parser_filter
    ]
    if not candidates:
        return None
    return min(
        candidates,
        key=lambda r: (
            -int(source_priority[r["source"]]),
            int(parser_tiebreak.get(r.get("parser_name") or "mostlyright_v1", 999)),
            r.get("source_received_at") or "",
            r.get("ingestion_id") or "",
        ),
    )

LIVE_V1 = MergePolicy(
    name="LIVE_V1",
    source_priority=MappingProxyType({"awc": 3, "iem": 2, "ghcnh": 1}),  # NCEI excluded per D-2.1-09
    parser_filter=frozenset({"mostlyright_v1"}),  # IEM-dual-row skipped per D-2.1-10
    parser_tiebreak=MappingProxyType({"mostlyright_v1": 0}),
    selector=first_reported_highest_priority_with_received_at_tiebreak,
)

def query_time_merge(rows: list[dict], *, policy: MergePolicy) -> list[dict]:
    # Group by natural key, then select one row per group using policy.selector
    groups: dict[tuple, list[dict]] = {}
    for r in rows:
        # Defensive guard (lifted from Sprint 2o-s1 spec lines 195-204)
        if (src := r.get("source")) and src not in policy.source_priority:
            raise ValueError(f"Unknown source {src!r} not in {policy.name} source_priority")
        key = (r["station_code"], r["observed_at"], r.get("observation_kind") or r.get("observation_type"))
        groups.setdefault(key, []).append(r)
    winners = [
        policy.selector(
            grp,
            source_priority=policy.source_priority,
            parser_filter=policy.parser_filter,
            parser_tiebreak=policy.parser_tiebreak,
        )
        for grp in groups.values()
    ]
    return [w for w in winners if w is not None]
```

```python
_SOURCE_PRIORITY = {"awc": 3, "iem": 2, "ghcnh": 1}

def merge_observations(rows: list[dict]) -> list[dict]:
    """Strict-> dedup; first row seen wins at equal priority."""
    best: dict[tuple, dict] = {}
    for row in rows:
        key = (row["station_code"], row["observed_at"], row["observation_type"])
        priority = _SOURCE_PRIORITY.get(row.get("source", ""), 0)
        if key not in best:
            best[key] = row
        else:
            existing_priority = _SOURCE_PRIORITY.get(best[key].get("source", ""), 0)
            if priority > existing_priority:
                best[key] = row
    return list(best.values())
```
