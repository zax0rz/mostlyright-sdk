       """Phase 2.1 migration CLI: pre-2.1 → silver-tier observation ledger.

       Walks the entire cache root, calls `cache._legacy_v014_to_v021_migration()`
       per legacy file. Idempotent. Climate path UNTOUCHED (D-2.1-09).

       Usage:
           python -m tradewinds.weather.cache_migration [--cache-root PATH] [--dry-run]
       """
       from __future__ import annotations
       import argparse
       import logging
       import os
       from pathlib import Path
       from . import cache

       _LOG = logging.getLogger(__name__)

       def migrate_to_v2(cache_root: Path | str | None = None, *, dry_run: bool = False) -> int:
           """Walk cache_root/observations/ and migrate each legacy parquet to observations_ledger/.

           Returns total rows migrated (or 0 in dry_run mode).
           Skips climate/ entirely (D-2.1-09).
           """
           if cache_root is None:
               cache_root = Path(os.environ.get("TRADEWINDS_CACHE_DIR", Path.home() / ".tradewinds" / "cache")) / "v1"
           else:
               cache_root = Path(cache_root)

           legacy_root = cache_root / cache.OBSERVATIONS_LEGACY_SUBDIR
           if not legacy_root.exists():
               _LOG.info("No legacy observations dir at %s — nothing to migrate", legacy_root)
               return 0

           total_rows = 0
           for legacy_path in sorted(legacy_root.rglob("*.parquet")):
               # Skip .legacy and .partial files
               if legacy_path.suffix != ".parquet" or legacy_path.name.endswith(".legacy"):
                   continue
               # Derive (station, year, month) from path: legacy_root/{station}/{YYYY}/{MM}.parquet
               try:
                   month = int(legacy_path.stem)
                   year = int(legacy_path.parent.name)
                   station = legacy_path.parent.parent.name
               except (ValueError, IndexError):
                   _LOG.warning("Skipping unparseable path: %s", legacy_path)
                   continue
               ledger_path = cache._ledger_path(cache_root, station, year, month)
               if ledger_path.exists():
                   _LOG.info("Skip (ledger exists): %s", legacy_path)
                   continue
               if dry_run:
                   _LOG.info("[DRY RUN] Would migrate %s -> %s", legacy_path, ledger_path)
                   continue
               rows = cache._legacy_v014_to_v021_migration(legacy_path, ledger_path, station, year, month)
               total_rows += rows
               _LOG.info("Migrated %d rows: %s -> %s", rows, legacy_path, ledger_path)
           return total_rows

       def _main() -> int:
           parser = argparse.ArgumentParser(description="Migrate pre-2.1 tradewinds cache to silver-tier ledger.")
           parser.add_argument("--cache-root", type=Path, default=None,
                               help="Cache root override (default: $TRADEWINDS_CACHE_DIR or ~/.tradewinds/cache)")
           parser.add_argument("--dry-run", action="store_true", help="Print what would happen; modify nothing.")
           args = parser.parse_args()
           logging.basicConfig(level=logging.INFO, format="%(message)s")
           total = migrate_to_v2(args.cache_root, dry_run=args.dry_run)
           print(f"Migrated {total} rows" + (" (dry run)" if args.dry_run else ""))
           return 0

       if __name__ == "__main__":
           raise SystemExit(_main())
