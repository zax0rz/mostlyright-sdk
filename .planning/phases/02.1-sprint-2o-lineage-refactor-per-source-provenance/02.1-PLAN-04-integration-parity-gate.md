def _assert_byte_equivalent(actual: pd.DataFrame, expected: pd.DataFrame, fixture_name: str):
    assert actual.shape == expected.shape
    assert list(actual.columns) == list(expected.columns)
    assert actual.dtypes.equals(expected.dtypes)
    for col in actual.columns:
        a, e = actual[col], expected[col]
        if pd.api.types.is_numeric_dtype(a):
            mask_a, mask_e = a.isna(), e.isna()
            assert mask_a.equals(mask_e)
            np.testing.assert_array_equal(a[~mask_a].values, e[~mask_e].values)
        else:
            assert a.equals(e)

    Mitigates RESEARCH Risk 1: if the new merge engine drifts from v0.14.1 strict-`>`
    semantics on legacy-shape input (no source_received_at, no ingestion_id), the
    5-fixture parity test will fail. This pre-flight isolates the merge-engine

    # Inline copy of pre-Wave-2 strict-> logic for direct comparison.
    # DO NOT import from observations.py — that's the new shim; we want the OLD logic here.
    _SOURCE_PRIORITY_V014 = {"awc": 3, "iem": 2, "ghcnh": 1}

    def _legacy_dedup_rows(rows: list[dict]) -> list[dict]:
        best: dict[tuple, dict] = {}
        for row in rows:
            key = (row["station_code"], row["observed_at"], row.get("observation_type") or row.get("observation_kind"))
            priority = _SOURCE_PRIORITY_V014.get(row.get("source", ""), 0)
            if key not in best:
                best[key] = row
            else:
                existing_priority = _SOURCE_PRIORITY_V014.get(best[key].get("source", ""), 0)
                if priority > existing_priority:
                    best[key] = row
        return list(best.values())
