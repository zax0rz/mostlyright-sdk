    """Shared Hypothesis strategies for Phase 2.1+ tests.

    All datetime strategies constrained to [2018-01-01, 2027-12-31] UTC per CORE-08
    to avoid Hypothesis shrinking pathology on unbounded datetime ranges (Phase 2.1
    Risk 4).
    """
    from datetime import datetime, timezone
    from hypothesis import strategies as st

    TRADEWINDS_MIN_DATETIME = datetime(2018, 1, 1, tzinfo=timezone.utc)
    TRADEWINDS_MAX_DATETIME = datetime(2027, 12, 31, 23, 59, 59, tzinfo=timezone.utc)

    def tradewinds_datetime_strategy() -> st.SearchStrategy[datetime]:
        """UTC datetime in [2018-01-01, 2027-12-31] — CORE-08 compliant."""
        return st.datetimes(
            min_value=TRADEWINDS_MIN_DATETIME.replace(tzinfo=None),
            max_value=TRADEWINDS_MAX_DATETIME.replace(tzinfo=None),
            timezones=st.just(timezone.utc),
        )

    def rfc3339_microsecond_strategy() -> st.SearchStrategy[str]:
        """RFC3339 UTC string with microsecond precision — matches source_received_at regex."""
        return tradewinds_datetime_strategy().map(
            lambda d: d.isoformat(timespec='microseconds')
        )

    def rfc3339_second_strategy() -> st.SearchStrategy[str]:
        """RFC3339 UTC string with second precision — matches as_of_time."""
        return tradewinds_datetime_strategy().map(
            lambda d: d.isoformat(timespec='seconds')
        )

    SOURCE_ENUM = st.sampled_from(["awc", "iem", "ghcnh"])  # ncei reserved but never drawn in v0.1.0
    PARSER_NAME_ENUM = st.just("mostlyright_v1")  # only parser written in v0.1.0 per D-2.1-10

    STATION_CODE_ENUM = st.sampled_from(["KNYC", "KMDW", "KLAX", "KMIA", "KMSY"])

        """Generates a silver-tier row dict with all 9 lineage fields populated.

        Used by Wave 2 merge-policy property tests and Wave 3 cache tests.
        """
        return st.fixed_dictionaries({
            "station_code": STATION_CODE_ENUM,
            "observed_at": tradewinds_datetime_strategy().map(lambda d: d.isoformat()),
            "observation_type": OBSERVATION_KIND_ENUM,
            "observation_kind": OBSERVATION_KIND_ENUM,
