    Use this one-shot Python script (interactive or save to `scripts/_phase12_workflows_rewrite.py`):

    ```python
    from pathlib import Path
    import re

    SUBSTITUTIONS = [
        # 1. Package CLI args
        (re.compile(r'--package\s+tradewinds(-weather|-markets|\b)'), r'--package mostlyright\1'),
        # 2. pnpm filter args
        (re.compile(r'--filter\s+tradewinds\b'), '--filter mostlyright'),
        (re.compile(r'--filter\s+@tradewinds/'), '--filter @mostlyrightmd/'),
        # 3. Source-path globs
        (re.compile(r'packages/(core|weather|markets)/src/tradewinds/'), r'packages/\1/src/mostlyright/'),
        # 4. @tradewinds/* scoped package refs
        (re.compile(r'@tradewinds/'), '@mostlyrightmd/'),
        # 5. TRADEWINDS_TS_LIVE env var rename (drift-rotate-ts.yml)
        (re.compile(r'TRADEWINDS_TS_LIVE'), 'MOSTLYRIGHT_TS_LIVE'),
        # 6. PyPI URL refs
        (re.compile(r'pypi\.org/project/tradewinds(-weather|-markets)?/'), r'pypi.org/project/mostlyright\1/'),
        # 7. Comment refs to tradewinds.research / tradewinds.* identifiers
        # (these may be in any context — the import-line-restricted pattern won't catch them; do a guarded replace)
    ]

    PRESERVE = re.compile(r'TW_HOSTED_URL|github\.com/helloiamvu/tradewinds|monorepo-v0\.14\.1|mostlyright==0\.14\.1')

    def rewrite_file(path: Path) -> int:
        text = path.read_text(encoding='utf-8')
        original = text
        for pat, repl in SUBSTITUTIONS:
            text = pat.sub(repl, text)
        # Sweep remaining lowercase `tradewinds` tokens (e.g. in comments) — but skip preserve-list lines
        out_lines = []
        for line in text.splitlines(keepends=True):
            if PRESERVE.search(line):
                # Skip the preserve-list portions but still rewrite OTHER `tradewinds` tokens on the same line
                # For YAML/JSON comments + workflow files this corner is rare; safest: leave the line untouched if it contains a preserve token AND has `tradewinds`.
                out_lines.append(line)
            elif 'tradewinds' in line:
                # Sweep remaining lowercase `tradewinds` (not @tradewinds, not -tradewinds, not /tradewinds — those are SUBSTITUTIONS captured above)
                new = line.replace('tradewinds', 'mostlyright')
                out_lines.append(new)
            else:
                out_lines.append(line)
        new_text = ''.join(out_lines)
        if new_text != original:
            path.write_text(new_text, encoding='utf-8')
            return 1
        return 0

    for path in Path('.github/workflows').rglob('*.yml'):
        if rewrite_file(path):
            print(f'  {path}')

    # Also rewrite parity-trigger-paths.json
    rewrite_file(Path('.github/parity-trigger-paths.json'))
    ```
