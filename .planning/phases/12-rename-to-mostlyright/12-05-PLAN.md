    ```python
    from pathlib import Path
    import re

    PRESERVE_PATTERNS = [

    ]
    PRESERVE_RE = re.compile('|'.join(PRESERVE_PATTERNS))

    def rewrite_doc(path: Path) -> int:
        text = path.read_text(encoding='utf-8')
        # Token-by-token: split on 'tradewinds' occurrences; substitute only if not preserve-context.
        # Simpler: process line-by-line.
        lines = text.splitlines(keepends=True)
        n_changed = 0
        new_lines = []
        for line in lines:
            if PRESERVE_RE.search(line):
                # Line contains a preserve-list literal — surgical replace: only substitute `tradewinds` tokens NOT inside the preserved match
                # For safety, use a finditer-based approach to skip the preserved substring
                spans_to_skip = [m.span() for m in PRESERVE_RE.finditer(line)]
                # Find all `tradewinds` matches (case-sensitive)
                out = []
                i = 0
                for m in re.finditer(r'tradewinds', line):
                    s, e = m.span()
                    in_preserve = any(ps <= s < pe for ps, pe in spans_to_skip)
                    if in_preserve:
                        continue
                    out.append((s, e))
                if not out:
                    new_lines.append(line)
                    continue
                # Rewrite each non-preserved match
                result = []
                last = 0
                for s, e in out:
                    result.append(line[last:s])
                    result.append('mostlyright')
                    last = e
                result.append(line[last:])
                new = ''.join(result)
            else:
                new = line.replace('tradewinds', 'mostlyright')
            if new != line:
                n_changed += 1
            new_lines.append(new)
        if n_changed:
            path.write_text(''.join(new_lines), encoding='utf-8')
        return n_changed

    for md in list(Path('docs').rglob('*.md')) + [Path('README.md')]:
        n = rewrite_doc(md)
        if n: print(f'  {md}: {n} lines')
    ```
